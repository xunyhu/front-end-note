```js
// 代码 A
for (var i = 0; i < 3; i++) {
  setTimeout(() => console.log(i), 0);
}

// 代码 B
for (let i = 0; i < 3; i++) {
  setTimeout(() => console.log(i), 0);
}

// 代码 C

}
```
